+def render_diff_html(diff_text: str) -> str:
+    """Colour-code a unified diff for display."""
+    lines = diff_text.splitlines()
+    html_lines = []
+    for line in lines:
+        if line.startswith("+") and not line.startswith("+++"):
+            html_lines.append(f'<span class="diff-add">{line}</span>')
+        elif line.startswith("-") and not line.startswith("---"):
+            html_lines.append(f'<span class="diff-rem">{line}</span>')
+        elif line.startswith("@@") or line.startswith("---") or line.startswith("+++"):
+            html_lines.append(f'<span class="diff-meta">{line}</span>')
+        else:
+            html_lines.append(line)
+    return "\n".join(html_lines)
